    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // Находим поле ввода и все карточки квестов
            const searchInput = document.querySelector('.search-box input');
            const questCards = document.querySelectorAll('.quest-card');

            // Добавляем слушатель событий, который реагирует на каждый введенный символ
            searchInput.addEventListener('input', function() {
                // Получаем введенный текст, переводим в нижний регистр и убираем лишние пробелы
                const searchText = this.value.toLowerCase().trim();

                // Перебираем все карточки квестов
                questCards.forEach(card => {
                    // Находим заголовок (название) внутри текущей карточки
                    const titleElement = card.querySelector('h3');
                    
                    if (titleElement) {
                        const titleText = titleElement.textContent.toLowerCase();
                        
                        // Если название квеста содержит введенный текст, показываем карточку, иначе - скрываем
                        if (titleText.includes(searchText)) {
                            card.style.display = ''; // Возвращаем стандартное отображение (grid)
                        } else {
                            card.style.display = 'none'; // Скрываем карточку
                        }
                    }
                });
            });
        });
    </script>
